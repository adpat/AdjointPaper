\section{Other highway/control models}

\frame{
  \setbeamercovered{transparent}
	\frametitle{Other highway/control models}
  \begin{itemize}
    \item<1-> Network CTM~\cite{daganzo1995cell}
    \begin{itemize}
			\item Exact Godunov discretization of LWR with triangular fundamental diagram
			\item only specifies $1\times2$, $2\times1$ junctions
		\end{itemize}
    \item<2-> Link-node CTM with ramp metering problem as LP~\cite{Muralidharana}
    \begin{itemize}
			\item For $n\times m$ junctions
			\item Priorities are proportional to demand
			\item In the limit, the junction solver is not "self-similar". More on this later.
			\item Either a relaxation of the junction model is necessary, or can equivalently be expressed as VSL
			\item Open-loop, can be put in MPC framework.
		\end{itemize}    
		\item<3-> Papageorgiou ALINEA controller~\cite{Papageorgiou1991}
    \begin{itemize}
			\item Closed-loop
			\item Only uses state estimation (no forward-sim model)
			\item Local policy (single ramp), extensions (HERO) for multi-ramp using heuristics
		\end{itemize}
  \end{itemize}
}
